def read_gmt(gmt_file):
    with open(gmt_file, 'r') as f:
        id_to_genes = dict()
        id_to_name = dict()
        for line in f.readlines():
            l = line.strip().split('\t')
            name = l[0]
            GOID = l[1]
            if len(l)>2:
                id_to_genes[GOID] = l[2:]
            else:
                id_to_genes[GOID] = []
            id_to_name[GOID] = name
    return(id_to_genes,id_to_name)

    freq = np.array(pd.cut(x, bins=100).value_counts())
    prob = freq/sum(freq)
    prob = prob[prob>0]
    log_prob = np.log2(prob)
    entropy = -np.dot(prob,log_prob)
